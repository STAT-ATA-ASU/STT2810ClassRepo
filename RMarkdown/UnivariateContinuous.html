> sims <- 10000
> nj <- rnorm(sims, 100, 15)
> ej <- rexp(sims, 0.01)
> library(ggplot2)
> DF <- data.frame(x = nj)
> p <- ggplot(data = DF, aes(x = x))
> p +  geom_histogram(fill = "lightblue") + 
+   theme_bw()

> p + geom_density(fill = "pink") + 
+   theme_bw()

The mean is 99.8497174 and the standard deviation is 15.03552.
